<template>
  <div class="event-card" v-if="isValid">
    <div class="event-header">
      <h2>{{ event.title || 'Untitled event' }}</h2>
      <span class="event-category">{{ event.category || 'Uncategorized' }}</span>
    </div>

    <div class="event-details">
      <p><strong>Date:</strong> {{ event.date || 'TBD' }}</p>
      <p><strong>Time:</strong> {{ event.time || 'TBD' }}</p>
      <p><strong>Location:</strong> {{ event.location || 'TBD' }}</p>
    </div>

    <div class="registration-info">
      <h3>Registration</h3>
      <p>
        {{ event.registrationStart || 'N/A' }}
        -
        {{ event.registrationEnd || 'N/A' }}
      </p>
    </div>

    <button
      class="register-btn"
      :disabled="!event.id"
      @click="$emit('open-registration', event.id)"
    >
      Register
    </button>

    <button
      class="details-btn"
      :disabled="!event.id"
      @click="$emit('open-details', event.id)"
    >
      View Details
    </button>
  </div>

  <!-- Optional: show nothing or a fallback if invalid -->
  <div v-else class="event-card invalid">
    Invalid event data
  </div>
</template>

<script>
export default {
  name: 'EventCard',
  props: {
    event: {
      type: Object,
      required: true,
      validator(value) {
        if (!value || typeof value !== 'object') return false

        const hasRequiredStrings =
          typeof value.title === 'string' &&
          typeof value.category === 'string'

        const hasId = value.id !== undefined && value.id !== null

        return hasRequiredStrings && hasId
      }
    }
  },
  computed: {
    isValid() {
      return !!this.event &&
        typeof this.event.title === 'string' &&
        typeof this.event.category === 'string'
    }
  }
}
</script>
import { createRouter, createWebHistory } from 'vue-router'

const router = createRouter({
  history: createWebHistory(),
  routes: [
    { path: '/events', component: () => import('./views/EventsList.vue') },
    { path: '/events/:id', component: () => import('./views/EventDetail.vue') },
    { path: '/events/registration/:id', component: () => import('./views/EventRegistration.vue') },

    // Redirect root
    { path: '/', redirect: '/events' },

    // Catch-all 404
    { path: '/:pathMatch(.*)*', component: () => import('./views/NotFoundPage.vue') }
  ]
})

// Optional: ignore only redundant navigation errors (Vue Router 4)
const originalPush = router.push
router.push = function (location) {
  return originalPush.call(this, location).catch(err => {
    // Ignore NavigationDuplicated-like errors
    if (err && err.name === 'NavigationDuplicated') return
    // Re-throw other errors so you see real routing issues
    throw err
  })
}

export default router

<template>
  <div class="event-list">
    <EventCard
      v-for="event in paginatedEvents"
      :key="event.id"
      :event="event"
      v-memo="[event.id]"
      @open-details="openDetails"
      @open-registration="openRegistration"
    />

    <div class="pagination">
      <button @click="prevPage" :disabled="page === 1">Prev</button>
      <span>Page {{ page }}</span>
      <button @click="nextPage" :disabled="endIndex >= events.length">Next</button>
    </div>
  </div>
</template>

<script>
import { defineAsyncComponent } from 'vue'

export default {
  name: 'EventList',

  components: {
    EventCard: defineAsyncComponent(() => import('./EventCard.vue'))
  },

  props: {
    events: {
      type: Array,
      required: true
    }
  },

  data() {
    return {
      page: 1,
      pageSize: 20
    }
  },

  computed: {
    startIndex() {
      return (this.page - 1) * this.pageSize
    },
    endIndex() {
      return this.startIndex + this.pageSize
    },
    paginatedEvents() {
      return this.events.slice(this.startIndex, this.endIndex)
    }
  },

  methods: {
    nextPage() {
      if (this.endIndex < this.events.length) this.page++
    },
    prevPage() {
      if (this.page > 1) this.page--
    },
    openDetails(id) {
      this.$router.push(`/events/${id}`)
    },
    openRegistration(id) {
      this.$router.push(`/events/registration/${id}`)
    }
  }
}
</script>
