<template>
  <div class="event-card" v-if="isEventValid">
    <div class="event-header">
      <h2>{{ event.title }}</h2>
      <span class="event-category">{{ event.category }}</span>
    </div>

    <div class="event-details">
      <p><strong>Date:</strong> {{ event.date || 'TBD' }}</p>
      <p><strong>Time:</strong> {{ event.time || 'TBD' }}</p>
      <p><strong>Location:</strong> {{ event.location || 'TBD' }}</p>
    </div>

    <div class="registration-info">
      <h3>Registration</h3>
      <p>
        {{ event.registrationStart || 'N/A' }} -
        {{ event.registrationEnd || 'N/A' }}
      </p>
    </div>

    <button
      class="register-btn"
      :disabled="!event.id || !isRegistrationOpen"
      @click="$emit('open-registration', event.id)"
    >
      Register
    </button>

    <!-- Registration Form -->
    <form @submit.prevent="submitRegistration">
      <div class="form-group">
        <label>First Name</label>
        <input
          v-model="form.firstName"
          type="text"
          @blur="validateField('firstName')"
        />
        <p v-if="errors.firstName" class="error">{{ errors.firstName }}</p>
      </div>

      <div class="form-group">
        <label>Last Name</label>
        <input
          v-model="form.lastName"
          type="text"
          @blur="validateField('lastName')"
        />
        <p v-if="errors.lastName" class="error">{{ errors.lastName }}</p>
      </div>

      <div class="form-group">
        <label>Email Address</label>
        <input
          v-model="form.email"
          type="email"
          @blur="validateField('email')"
        />
        <p v-if="errors.email" class="error">{{ errors.email }}</p>
      </div>

      <button type="submit" :disabled="!isFormValid">
        Submit Registration
      </button>
    </form>
  </div>
</template>

<script>
export default {
  name: 'EventCard',

  props: {
    event: {
      type: Object,
      required: true,
      validator(value) {
        return (
          typeof value.title === 'string' &&
          typeof value.category === 'string' &&
          value.id != null
        )
      }
    }
  },

  data() {
    return {
      form: {
        firstName: '',
        lastName: '',
        email: ''
      },
      errors: {}
    }
  },

  computed: {
    isEventValid() {
      return !!this.event
    },

    isFormValid() {
      return (
        this.form.firstName &&
        this.form.lastName &&
        this.validateEmail(this.form.email) &&
        Object.keys(this.errors).length === 0
      )
    },

    isRegistrationOpen() {
      const now = new Date()
      return (
        now >= new Date(this.event.registrationStart) &&
        now <= new Date(this.event.registrationEnd)
      )
    }
  },

  methods: {
    validateField(field) {
      const value = this.form[field]

      if (!value) {
        this.errors[field] = 'This field is required'
        return
      }

      if (field === 'email' && !this.validateEmail(value)) {
        this.errors.email = 'Invalid email format'
        return
      }

      delete this.errors[field]
    },

    validateEmail(email) {
      return /\S+@\S+\.\S+/.test(email)
    },

    submitRegistration() {
      if (!this.isFormValid) return
      this.$emit('submit-registration', { ...this.form })
    }
  }
}
</script>

<style scoped>
.event-card {
  border: 1px solid #ccc;
  padding: 1rem;
}
.error {
  color: red;
  font-size: 0.85rem;
}
</style>
